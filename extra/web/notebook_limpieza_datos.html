# ── 1.2 Limpieza ─────────────────────────────────────────────
df_clean = df.copy()

# 1. Tipado correcto de fechas
for col in ["fecha_inicio", "fecha_hecho"]:
    df_clean[col] = pd.to_datetime(df_clean[col], errors="coerce")

# 2. Normalizar alcaldías — campo texto libre tiene muchas variantes
df_clean["alcaldia_norm"] = (
    df_clean["alcaldia_hecho"]
    .str.strip().str.upper()
    .map(ALCALDIAS_MAP)
    .fillna(df_clean["alcaldia_hecho"])  # fallback al original
)

# 3. Hora numérica — extraer solo la parte HH:MM
df_clean["hora_inicio_clean"] = (
    df_clean["hora_inicio"].str[:5]  # "HH:MM:SS" → "HH:MM"
)

# 4. Mes → número ordinal para ordenamiento
MESES_NUM = {"Enero":1,"Febrero":2,"Marzo":3,"Abril":4,
             "Mayo":5,"Junio":6,"Julio":7,"Agosto":8,
             "Septiembre":9,"Octubre":10,"Noviembre":11,"Diciembre":12}
df_clean["mes_num"] = df_clean["mes_inicio"].map(MESES_NUM)

# 5. Descartar registros sin coordenadas
n_sin_coords = df_clean["latitud"].isna().sum()
df_geo = df_clean.dropna(subset=["latitud", "longitud"]).copy()
print(f"Registros descartados sin coords: {n_sin_coords:,} ({n_sin_coords/len(df_clean)*100:.1f}%)")
print(f"Registros con coords: {len(df_geo):,}")

# ── 3.2 Enriquecer y limpiar ─────────────────────────────────

# 1. Decodificar sistema de transporte desde zone_id
#    Formato: "XXXXX-NOMBRE_PARADA" donde prefijo numérico identifica la ruta
def inferir_sistema(zone_id):
    z = str(zone_id).upper()
    if z.startswith("0300"): return "Metrobús"
    if z.startswith("05"):   return "Metro"
    if z.startswith("04"):   return "RTP"
    if z.startswith("06"):   return "Tren Ligero"
    if z.startswith("07"):   return "Cablebús"
    if z.startswith("08"):   return "Trolebús"
    return "Otro"

df_stops["sistema"] = df_stops["zone_id"].apply(inferir_sistema)

# 2. Accesibilidad legible
df_stops["accesible"] = df_stops["wheelchair_boarding"].map({
    0: "sin_info", 1: "accesible", 2: "inaccesible"
})

# 3. Filtrar paradas fuera de CDMX bbox
n_orig = len(df_stops)
df_stops = df_stops[
    (df_stops.stop_lat.between(BBOX_CDMX["lat_min"], BBOX_CDMX["lat_max"])) &
    (df_stops.stop_lon.between(BBOX_CDMX["lon_min"], BBOX_CDMX["lon_max"]))
].copy()
print(f"Paradas filtradas fuera de bbox: {n_orig - len(df_stops)}")

# 4. GeoDataFrame → UTM 14N
gdf_stops = gpd.GeoDataFrame(
    df_stops[["stop_id", "stop_name", "sistema", "accesible", "stop_lat", "stop_lon"]],
    geometry=gpd.points_from_xy(df_stops.stop_lon, df_stops.stop_lat),
    crs=WGS84
).to_crs(UTM14)

RI-6 — Infraestructura Física por Colonia (INEGI) 1,814 polígonos · sin nulos · EPSG:32614

Shapefile con índice de rezago en infraestructura física (RI-6) calculado por INEGI a nivel de colonia. Ya viene en UTM 14N. Las columnas más importantes para el proyecto son el alumbrado (AlumP_sum, r_AlumP_Mn) y el score compuesto (S_RI_6, C_RI_6).

Nota sobre AlumP_sum. No representa el número de luminarias sino una puntuación ponderada de alumbrado por manzana, sumada a nivel colonia. Un valor de 0 indica colonias sin ningún equipamiento registrado — hay 20 colonias en esa condición.

# Variables originales que conservamos
COLS_INFRA = [

    "colonia",         # nombre de colonia
    "alcaldia",        # alcaldía original
    # Alumbrado
    "AlumP_sum",       # score acumulado de alumbrado
    "ALUMP_mean",      # score promedio por manzana
    "r_AlumP_Mn",      # ranking 1(mejor)–5(peor)
    # Servicios básicos
    "P_PorcElec",      # % viviendas con electricidad
    "P_AguaPot",       # % viviendas con agua potable
    # Score compuesto
    "S_RI_6",          # score continuo (mayor = mayor rezago)
    "C_RI_6",          # clase 1–5 (5 = mayor rezago)
    "geometry",        # polígono colonia — CRS: UTM 14N
]

gdf_infra_s = gdf_infra[COLS_INFRA].copy()

# 1. Normalizar nombre de alcaldía
gdf_infra_s["alcaldia_norm"] = (
    gdf_infra_s["alcaldia"].str.strip().str.upper().map(ALCALDIAS_MAP)
)

# 2. Variable derivada: vulnerabilidad_luz [0.0 – 1.0]
#    Transforma r_AlumP_Mn (1=mejor,5=peor) a escala normalizada
#    donde 0.0 = colonia con mejor alumbrado, 1.0 = peor
gdf_infra_s["vulnerabilidad_luz"] = np.where(
    gdf_infra_s["r_AlumP_Mn"] > 0,
    (gdf_infra_s["r_AlumP_Mn"] - 1) / 4,   # escala 1-5 → 0.0-1.0
    1.0                                      # r_AlumP_Mn == 0 → máxima vulnerabilidad
)

# 3. Variable derivada: rezago_infra [0.0 – 1.0]
#    Normaliza S_RI_6 al rango observado
mn, mx = gdf_infra_s["S_RI_6"].min(), gdf_infra_s["S_RI_6"].max()
gdf_infra_s["rezago_infra"] = (gdf_infra_s["S_RI_6"] - mn) / (mx - mn)

print("Variables derivadas:")

con.execute(f"CREATE VIEW stops  AS SELECT * FROM read_parquet('{SILVER}/stops_gtfs.parquet')")

# ── 5.2 Resumen estadístico con DuckDB ───────────────────────
resumen_fgj = con.execute("""
    SELECT
        relevancia,
        COUNT(*)             AS n_registros,
        COUNT(DISTINCT mes_num) AS meses_cubiertos,
        COUNT(DISTINCT alcaldia_norm) AS alcaldias
    FROM fgj
    GROUP BY relevancia
    ORDER BY n_registros DESC
""").df()

print("=== FGJ por relevancia ===")
print(resumen_fgj.to_string(index=False))

# ── 5.3 Exportar Gold: tabla de conteos por alcaldía ─────────
gold_alcaldia = con.execute("""
    SELECT
        f.alcaldia_norm,
        COUNT(CASE WHEN f.relevancia = 'peatón' THEN 1 END) AS delitos_peaton,
        COUNT(CASE WHEN f.relevancia = 'transporte' THEN 1 END) AS delitos_transporte,
        COUNT(*) AS delitos_total
    FROM fgj f
    WHERE f.alcaldia_norm IS NOT NULL
    GROUP BY f.alcaldia_norm
    ORDER BY delitos_peaton DESC
""").df()

print("\n=== Delitos por alcaldía ===")
print(gold_alcaldia.to_string(index=False))

# Exportar
